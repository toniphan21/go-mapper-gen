## Configuration

Let set up an empty golang project without any dependency.

```go.mod
module github.com/toniphan21/go-mapper-gen/config

go 1.25
```

Given that you have 2 structs in your `domain`:

```go
// file: domain/entity.go

package domain

type User struct {
	ID    string
	Name  string
	Email string
	Age   int
}

type Address struct {
	ID          string
	HouseNumber string
	Street      string
	City        string
	State       string
	Country     string
}
```

and 2 structs in `rest` packages

```go
// file: rest/message.go
//go:generate go run github.com/toniphan21/go-mapper-gen/cmd/generator

package rest

type UserResponse struct {
	ID    string
	Name  string
	Email string
	Age   int
}

type AddressResponse struct {
	ID          string
	HouseNumber string
	Street      string
	City        string
	State       string
	Country     string
}
```

### Change functions name

By default, the function converts source to target name is `To{TargetStructName}`,
function convert source from target is `From{TargetStructName}`, we can customize them
on 2 levels: package and individual struct.

```pkl
packages {
	["github.com/toniphan21/go-mapper-gen/config/rest"] {
		source_pkg = "github.com/toniphan21/go-mapper-gen/config/domain"

		// these configs are package level, applied to all structs
		source_to_target_function_name = "Make{TargetStructName}"
		source_from_target_function_name = "To{SourceStructName}"

		structs {
			["UserResponse"] {
				source_struct_name = "User"

				// this config is per struct level, it overrides the package's level value
				source_from_target_function_name = "ConvertTo{SourceStructName}"
			}

			["AddressResponse"] {
				source_struct_name = "Address"

				// this config is per struct level, it overrides the package's level value
				source_to_target_function_name = "Create{TargetStructName}"
			}
		}
	}
}
```

```go

package rest

import domain "github.com/toniphan21/go-mapper-gen/config/domain"

type iMapper interface {
	// CreateAddressResponse converts a domain.Address value into a AddressResponse
	// value.
	CreateAddressResponse(in domain.Address) AddressResponse

	// ToAddress converts a AddressResponse value into a domain.Address value.
	ToAddress(in AddressResponse) domain.Address

	// MakeUserResponse converts a domain.User value into a UserResponse value.
	MakeUserResponse(in domain.User) UserResponse

	// ConvertToUser converts a UserResponse value into a domain.User value.
	ConvertToUser(in UserResponse) domain.User
}

func new_iMapper() iMapper {
	return &iMapperImpl{}
}

type iMapperImpl struct{}

func (m *iMapperImpl) CreateAddressResponse(in domain.Address) AddressResponse {
	var out AddressResponse

	out.ID = in.ID
	out.HouseNumber = in.HouseNumber
	out.Street = in.Street
	out.City = in.City
	out.State = in.State
	out.Country = in.Country

	return out
}

func (m *iMapperImpl) ToAddress(in AddressResponse) domain.Address {
	var out domain.Address

	out.ID = in.ID
	out.HouseNumber = in.HouseNumber
	out.Street = in.Street
	out.City = in.City
	out.State = in.State
	out.Country = in.Country

	return out
}

func (m *iMapperImpl) MakeUserResponse(in domain.User) UserResponse {
	var out UserResponse

	out.ID = in.ID
	out.Name = in.Name
	out.Email = in.Email
	out.Age = in.Age

	return out
}

func (m *iMapperImpl) ConvertToUser(in UserResponse) domain.User {
	var out domain.User

	out.ID = in.ID
	out.Name = in.Name
	out.Email = in.Email
	out.Age = in.Age

	return out
}

var _ iMapper = (*iMapperImpl)(nil)
```
